\begin{frame}{Optimal Control: Discrete-Time Systems}
    Given the time points $\{ t_0, \ldots, t_N\}$, with interarrivals $\tau_k, k \in \{ 0, \ldots, N \} $, the discrete-time state space equations are:

    \bigskip
    \centering
    \(
        \left\{
            \begin{array}{rcl}
                x_{k+1} &=& \overline{A}_k \,x_k + \overline{B}_k \\
                x\,(0)      &=& x_0
            \end{array}
        \right.
    \)

    \bigskip
    \begin{itemize}
        \item $\overline{A}_k = \Phi(\tau_k), \qquad \Phi(\tau) = e^{A \tau_k}$
        \item $\overline{B}_k = {\Gamma}(\tau_k), \qquad {\Gamma}(\tau) = \int_0^{\tau}\, e^{A (\tau - t)}\, dt\, B$
    \end{itemize}
\end{frame}
